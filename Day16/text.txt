#include<stdio.h>
#include<unistd.h>
#include <stdbool.h>
int main()
{
	int num1;
	int num2;
	int num3;
	
	printf("Enter the first number:");
    scanf("\n%d",&num1);
	printf("Enter the second number:");
	scanf("\n%d",&num2);
	printf("Enter the third number:");
	scanf("\n%d",&num3);
			
	int f=fork();
	if(f==0)
	{
		printf("Child of A(B)\n");
		printf("Parent Id:%d\n",getppid());
		int factorial=1;
		for(int i=1;i<=num1;i++)
		{
			factorial=factorial*i;
		}
		printf("\nFactorial of first number:%d",factorial);
	}
	
	else
	{
		int f1=fork();
		if(f1==0)
		{
			printf("\nChild of A(C)\n");
			printf("Parent Id:%d\n",getppid());
			
			int f2=fork();
			if(f2==0)
			{
				printf("\nChild of C(D)\n");
			    printf("Parent Id:%d\n",getppid());
				
				int a = 0, b = 1;
                printf("\nFibonacci Series: %d, %d", a, b);
                for (int i = 3; i <= num2; i++) {
                    int c = a + b;
                    printf(", %d", c);
                    a = b;
                    b = c;
                }
                printf("\n");
			}
			else
			{
				int f3=fork();
				if(f3==0)
				{
					printf("Child of C(E)\n");
			        printf("Parent Id:%d\n",getppid());
					
					bool isPrime = (num3 > 1);
                    for (int i = 2; i * i <= num3; i++) {
                        if (num3 % i == 0) {
                            isPrime = false;
                            break;
                        }
                    }
                    
                    if (isPrime) {
                        printf("The number %d is prime.\n", num3);
                    } else {
                        printf("The number %d is not prime.\n", num3);
                    }
				}
				
				else
				{
					printf("Parent (C)\n");
		            printf("Id:%d\n",getpid());
				}
			}
			
			
		}
		
		else
		{
			
			printf("Parent (A)\n");
		    printf("Id:%d\n",getpid());
		}
	}
	return 0;
}
