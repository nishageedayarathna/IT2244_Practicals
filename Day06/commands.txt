(07)Create simple calculator using "expr" command same as the Question03

[2021ict09@fedora ~]$ touch calculator_expr.sh
[2021ict09@fedora ~]$ vi calculator_expr.sh

echo "Enter the first number:"
read num1                          # Read first number

echo "Enter the second number:"
read num2                          # Read second number

# Use expr for arithmetic operations
sum=$(expr $num1 + $num2)                       
sub=$(expr $num1 - $num2)  
div=$(expr $num1 / $num2)  
mul=$(expr $num1 * $num2)  

# Display results
echo "Summation:$sum"
echo "Subtraction:$sub"
echo "Division:$div"
echo "Multiplication:$mul"
